---
title: "Create Bulk Phone Calls"
---

Creates multiple outbound phone calls in a single request using your configured assistant.

<Info>
  For full setup and authentication options, see the Installation guide: [/api-reference/introduction](/api-reference/introduction).
</Info>

<ParamField body="clientCfg" type="ConnectionConfig" required>
  Configuration for the client connection.
</ParamField>

<ParamField body="assistant" type="AssistantDefinition" required>
  Assistant configuration object.
</ParamField>

<ParamField body="params" type="{ toNumber: string; fromNumber?: string; args?: Map<string, google_protobuf_any_pb.Any>; options?: Map<string, google_protobuf_any_pb.Any>; metadata?: Map<string, google_protobuf_any_pb.Any>; }[]" required>
  Array of call parameter objects. Each object contains the destination and optional configuration for an individual call.
</ParamField>

<Fields>
  <ResponseField name="toNumber" type="string" required>
    Destination phone number in E.164 format (for example, +14155550123).
  </ResponseField>
  <ResponseField name="fromNumber" type="string">
    Caller ID phone number to originate the call from (E.164 format). If omitted, the system default will be used.
  </ResponseField>
  <ResponseField name="args" type="Map<string, google_protobuf_any_pb.Any>">
    Additional arguments passed to the call as a key-value map.
  </ResponseField>
  <ResponseField name="options" type="Map<string, google_protobuf_any_pb.Any>">
    Call options for runtime behavior and configuration.
  </ResponseField>
  <ResponseField name="metadata" type="Map<string, google_protobuf_any_pb.Any>">
    Arbitrary metadata to associate with the call for tracking and analytics.
  </ResponseField>
</Fields>

### Usage

<CodeGroup>

```ts TypeScript
import { CreateBulkPhoneCall, AssistantDefinition, type ConnectionConfig } from 'rapida-nodejs';

const clientCfg: ConnectionConfig = { /* ... */ };

const assistant = new AssistantDefinition();
assistant.setAssistantid(assistantId);
assistant.setVersion('latest');

const params = [
  {
    toNumber: `+14155550123`,
    fromNumber: `+14155559876`,
    args: new Map([["campaignId", { value: "fall-2024" }]]),
    options: new Map(),
    metadata: new Map([["source", { value: "crm" }]]),
  },
  { toNumber: `+441234567890` },
];

const res = await CreateBulkPhoneCall(clientCfg, assistant, params);
console.log('Bulk phone calls requested:', res);
```

### Response

The `CreateBulkPhoneCall` request returns a `CreateBulkPhoneCallResponse`.

| Field | Type | Description |
|-------|------|-------------|
| `code` | `number` | Numeric status code for the operation |
| `success` | `boolean` | Indicates whether the operation was successful |
| `data` | `AssistantConversation[]`, `undefined` | When successful, contains per-request results (may be undefined) |
| `error` | `Error` | When unsuccessful, contains error information (may be undefined) |
